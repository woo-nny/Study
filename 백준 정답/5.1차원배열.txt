import sys

N = int(sys.stdin.readline())
res =[]
sum =0
count = 0

    num = map(int,sys.stdin.readline().split())
    num=list(num)
    for b in range(1,len(num),1):
        sum += num[b]
    avr=sum/num[0]
    for b in range(1,len(num),1):
        if(num[b]-avr > 0):
            count += 1
    res.append((count/num[0])*100)
